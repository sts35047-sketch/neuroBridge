<script>
(function () {

    /* ═══════════════════════════════════════════════════════
       RENDERER + SCENE + CAMERA
    ═══════════════════════════════════════════════════════ */
    const canvas = document.getElementById('neural-canvas');
    const renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: true });
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.setClearColor(0x000000, 0);

    const scene  = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(50, window.innerWidth / window.innerHeight, 0.1, 1200);
    camera.position.set(0, 12, 110);

    /* ═══════════════════════════════════════════════════════
       MOUSE
    ═══════════════════════════════════════════════════════ */
    const mouse  = { x: 0, y: 0 };
    const target = { x: 0, y: 0 };
    document.addEventListener('mousemove', e => {
        target.x = (e.clientX / window.innerWidth  - 0.5);
        target.y = (e.clientY / window.innerHeight - 0.5);
    });

    /* ═══════════════════════════════════════════════════════
       ROOT GROUP — whole brain rotates together
    ═══════════════════════════════════════════════════════ */
    const root = new THREE.Group();
    scene.add(root);

    /* ═══════════════════════════════════════════════════════
       COLOUR CONSTANTS
    ═══════════════════════════════════════════════════════ */
    const CYN  = 0x00e5ff;
    const TCYN = 0x0099bb;
    const VIO  = 0x7c3aed;
    const PUR  = 0xc084fc;
    const EMR  = 0x10b981;
    const WHT  = 0xffffff;
    const CRM  = 0xff2d55;

    /* ═══════════════════════════════════════════════════════
       HEMISPHERE BUILDER
       A brain hemisphere is approximated by a half-ellipsoid
       whose surface is covered with GYRUS RIDGE TUBES.
       Each gyrus is a CatmullRom tube that follows a curved
       path over the surface — giving the folded, wrinkled
       look of real cortex.
    ═══════════════════════════════════════════════════════ */

    /* Sample a point on an ellipsoid surface */
    function ellipsoidPt(a, b, c, phi, theta) {
        return new THREE.Vector3(
            a * Math.sin(phi) * Math.cos(theta),
            b * Math.cos(phi),
            c * Math.sin(phi) * Math.sin(theta)
        );
    }

    /* Build one gyrus as a tube following a surface path.
       axis: 'lat'=horizontal fold, 'lon'=vertical fold      */
    function addGyrus(group, a, b, c, fixedAngle, swingRange,
                      axis, segments, tubeR, color, opacity, offX) {
        const pts = [];
        for (let s = 0; s <= segments; s++) {
            const t = s / segments;
            let phi, theta;
            if (axis === 'lat') {           // horizontal wrinkle
                phi   = fixedAngle + Math.sin(t * Math.PI * 3) * 0.08;
                theta = swingRange[0] + t * (swingRange[1] - swingRange[0]);
            } else {                        // vertical / meridional wrinkle
                phi   = swingRange[0] + t * (swingRange[1] - swingRange[0]);
                theta = fixedAngle + Math.sin(t * Math.PI * 4) * 0.10;
            }
            const p = ellipsoidPt(a, b, c, phi, theta);
            p.x += offX;
            pts.push(p);
        }
        const curve = new THREE.CatmullRomCurve3(pts);
        const geo   = new THREE.TubeGeometry(curve, segments, tubeR, 7, false);
        const mat   = new THREE.MeshBasicMaterial({ color, transparent: true, opacity });
        const mesh  = new THREE.Mesh(geo, mat);
        group.add(mesh);
        return mesh;
    }

    /* ── LEFT HEMISPHERE ── */
    const leftHemi = new THREE.Group();
    root.add(leftHemi);

    const LA = 26, LB = 22, LC = 20;   // ellipsoid semi-axes
    const LOFF = -14;                    // shift left

    // Base shell — dim filled half-ellipsoid
    const lShellGeo = new THREE.SphereGeometry(1, 48, 32, 0, Math.PI);
    lShellGeo.scale(LA, LB, LC);
    leftHemi.add(new THREE.Mesh(lShellGeo,
        new THREE.MeshBasicMaterial({ color: 0x0d1a2e, transparent: true, opacity: 0.55 })));

    // Wireframe cage over shell
    leftHemi.add(new THREE.Mesh(lShellGeo.clone(),
        new THREE.MeshBasicMaterial({ color: CYN, transparent: true, opacity: 0.05, wireframe: true })));

    // Horizontal gyri (latitude folds) — left hemi
    const latAnglesL = [0.55, 0.75, 0.95, 1.15, 1.35, 1.55, 1.75, 1.95, 2.15, 2.35];
    latAnglesL.forEach((phi, i) => {
        const color  = i % 3 === 0 ? CYN : (i % 3 === 1 ? PUR : EMR);
        const op     = 0.30 + (i % 2) * 0.12;
        addGyrus(leftHemi, LA, LB, LC, phi, [0.05, Math.PI - 0.05],
                 'lat', 40, 0.55, color, op, LOFF);
    });

    // Vertical / meridional sulci — left hemi
    const lonAnglesL = [0.18, 0.45, 0.75, 1.05, 1.35, 1.65, 1.95, 2.25, 2.55, 2.80];
    lonAnglesL.forEach((theta, i) => {
        const color = i % 2 === 0 ? TCYN : VIO;
        addGyrus(leftHemi, LA, LB, LC, theta, [0.25, Math.PI - 0.2],
                 'lon', 32, 0.40, color, 0.22, LOFF);
    });

    // Deep sulcus lines — thinner, darker, diagonal
    [[0.62, [0.3, 2.1]], [1.10, [0.4, 2.5]], [1.58, [0.2, 2.8]], [2.0, [0.35, 2.6]]].forEach(([phi, sw]) => {
        addGyrus(leftHemi, LA * 0.97, LB * 0.97, LC * 0.97, phi, sw,
                 'lat', 28, 0.28, CYN, 0.18, LOFF);
    });

    leftHemi.position.x = LOFF;

    /* ── RIGHT HEMISPHERE ── */
    const rightHemi = new THREE.Group();
    root.add(rightHemi);

    const RA = 26, RB = 22, RC = 20;
    const ROFF = 14;

    const rShellGeo = new THREE.SphereGeometry(1, 48, 32, Math.PI, Math.PI);
    rShellGeo.scale(RA, RB, RC);
    rightHemi.add(new THREE.Mesh(rShellGeo,
        new THREE.MeshBasicMaterial({ color: 0x0d1a2e, transparent: true, opacity: 0.55 })));
    rightHemi.add(new THREE.Mesh(rShellGeo.clone(),
        new THREE.MeshBasicMaterial({ color: PUR, transparent: true, opacity: 0.05, wireframe: true })));

    const latAnglesR = [0.52, 0.72, 0.92, 1.12, 1.32, 1.52, 1.72, 1.92, 2.12, 2.32];
    latAnglesR.forEach((phi, i) => {
        const color = i % 3 === 0 ? PUR : (i % 3 === 1 ? CYN : VIO);
        const op    = 0.28 + (i % 2) * 0.14;
        addGyrus(rightHemi, RA, RB, RC, phi, [Math.PI + 0.05, Math.PI * 2 - 0.05],
                 'lat', 40, 0.55, color, op, ROFF);
    });

    const lonAnglesR = [3.45, 3.72, 4.02, 4.32, 4.62, 4.92, 5.22, 5.52, 5.82, 6.08];
    lonAnglesR.forEach((theta, i) => {
        const color = i % 2 === 0 ? VIO : TCYN;
        addGyrus(rightHemi, RA, RB, RC, theta, [0.25, Math.PI - 0.2],
                 'lon', 32, 0.40, color, 0.22, ROFF);
    });

    [[0.60, [Math.PI + 0.3, Math.PI * 2 - 0.3]],
     [1.08, [Math.PI + 0.4, Math.PI * 2 - 0.4]],
     [1.56, [Math.PI + 0.2, Math.PI * 2 - 0.25]],
     [2.00, [Math.PI + 0.35, Math.PI * 2 - 0.35]]].forEach(([phi, sw]) => {
        addGyrus(rightHemi, RA * 0.97, RB * 0.97, RC * 0.97, phi, sw,
                 'lat', 28, 0.28, PUR, 0.18, ROFF);
    });

    rightHemi.position.x = ROFF;

    /* ── INTER-HEMISPHERIC FISSURE LINE ── */
    const fissurePts = [];
    for (let i = 0; i <= 30; i++) {
        const t = i / 30;
        fissurePts.push(new THREE.Vector3(
            Math.sin(t * Math.PI) * 0.5,
            LB * Math.cos(t * Math.PI * 0.75 + 0.3) * 0.9,
            LC * (t - 0.5) * 1.6
        ));
    }
    const fissureGeo = new THREE.BufferGeometry().setFromPoints(fissurePts);
    root.add(new THREE.Line(fissureGeo,
        new THREE.LineBasicMaterial({ color: CYN, transparent: true, opacity: 0.35 })));

    /* ═══════════════════════════════════════════════════════
       CORPUS CALLOSUM — the connecting bridge between hemispheres
       Arch-shaped tube running front-to-back at the base
    ═══════════════════════════════════════════════════════ */
    const ccPts = [];
    for (let i = 0; i <= 24; i++) {
        const t = (i / 24) * Math.PI;
        ccPts.push(new THREE.Vector3(
            Math.cos(t) * 18,
            -4 + Math.sin(t) * 5,
            0
        ));
    }
    const ccCurve = new THREE.CatmullRomCurve3(ccPts);
    root.add(new THREE.Mesh(
        new THREE.TubeGeometry(ccCurve, 30, 1.8, 10, false),
        new THREE.MeshBasicMaterial({ color: WHT, transparent: true, opacity: 0.22 })
    ));
    // Wireframe overlay on CC
    root.add(new THREE.Mesh(
        new THREE.TubeGeometry(ccCurve, 30, 1.9, 10, false),
        new THREE.MeshBasicMaterial({ color: CYN, transparent: true, opacity: 0.08, wireframe: true })
    ));

    /* ═══════════════════════════════════════════════════════
       BRAINSTEM — tapered cylinder dropping below cerebrum
    ═══════════════════════════════════════════════════════ */
    const bsGroup = new THREE.Group();
    bsGroup.position.set(0, -22, 8);
    root.add(bsGroup);

    // Pons (top)
    const ponsMesh = new THREE.Mesh(
        new THREE.CylinderGeometry(5.5, 6, 8, 16),
        new THREE.MeshBasicMaterial({ color: VIO, transparent: true, opacity: 0.30 })
    );
    ponsMesh.position.y = -2;
    bsGroup.add(ponsMesh);
    bsGroup.add(new THREE.Mesh(
        new THREE.CylinderGeometry(5.6, 6.1, 8, 16),
        new THREE.MeshBasicMaterial({ color: PUR, transparent: true, opacity: 0.08, wireframe: true })
    ));

    // Medulla oblongata (tapering downward)
    const medGeo = new THREE.CylinderGeometry(3.5, 2.2, 10, 12);
    const medMesh = new THREE.Mesh(medGeo,
        new THREE.MeshBasicMaterial({ color: TCYN, transparent: true, opacity: 0.28 }));
    medMesh.position.y = -11;
    bsGroup.add(medMesh);

    // Brainstem surface lines
    for (let i = 0; i < 8; i++) {
        const theta = (i / 8) * Math.PI * 2;
        const bsLinePts = [
            new THREE.Vector3(5.2 * Math.cos(theta), 2, 5.2 * Math.sin(theta)),
            new THREE.Vector3(4.5 * Math.cos(theta), -6, 4.5 * Math.sin(theta)),
            new THREE.Vector3(3.2 * Math.cos(theta), -16, 3.2 * Math.sin(theta)),
        ];
        bsGroup.add(new THREE.Line(
            new THREE.BufferGeometry().setFromPoints(bsLinePts),
            new THREE.LineBasicMaterial({ color: CYN, transparent: true, opacity: 0.20 })
        ));
    }

    /* ═══════════════════════════════════════════════════════
       CEREBELLUM — posterior, two lobes of folded mini-ridges
    ═══════════════════════════════════════════════════════ */
    const cbGroup = new THREE.Group();
    cbGroup.position.set(0, -18, -18);
    root.add(cbGroup);

    // Two cerebellar hemispheres
    [-9, 9].forEach((ox, side) => {
        const cbGeo = new THREE.SphereGeometry(1, 28, 20);
        cbGeo.scale(9, 7, 8);
        cbGroup.add(new THREE.Mesh(cbGeo,
            new THREE.MeshBasicMaterial({ color: VIO, transparent: true, opacity: 0.22 })));
        cbGroup.children[cbGroup.children.length - 1].position.x = ox;

        // Tight horizontal folia lines
        for (let f = 0; f < 9; f++) {
            const phi = 0.5 + f * 0.22;
            const foliaPts = [];
            const tRange = side === 0
                ? [0.05, Math.PI - 0.05]
                : [Math.PI + 0.05, Math.PI * 2 - 0.05];
            for (let k = 0; k <= 20; k++) {
                const theta = tRange[0] + (k / 20) * (tRange[1] - tRange[0]);
                foliaPts.push(new THREE.Vector3(
                    9 * Math.sin(phi) * Math.cos(theta) + ox,
                    7 * Math.cos(phi),
                    8 * Math.sin(phi) * Math.sin(theta)
                ));
            }
            cbGroup.add(new THREE.Line(
                new THREE.BufferGeometry().setFromPoints(foliaPts),
                new THREE.LineBasicMaterial({ color: side === 0 ? CYN : PUR, transparent: true, opacity: 0.28 })
            ));
        }
    });

    /* ═══════════════════════════════════════════════════════
       NEURAL PATHWAYS — long fibre tracts as graceful tubes
       connecting cortex → brainstem like white matter tracts
    ═══════════════════════════════════════════════════════ */
    const tractDefs = [
        // Corticospinal — left and right
        { pts: [[-10, -2, 12], [-8, -8, 8], [-5, -15, 6], [-2, -22, 8]],  color: CYN,  r: 0.55, op: 0.45 },
        { pts: [[ 10, -2, 12], [ 8, -8, 8], [ 5, -15, 6], [ 2, -22, 8]],  color: PUR,  r: 0.55, op: 0.45 },
        // Corpus callosum fibres fanning out
        { pts: [[-18, -4, 0], [-10, 2, 5],  [-2, 5, 8],   [ 2, 5, 8]],    color: WHT,  r: 0.40, op: 0.20 },
        { pts: [[ 18, -4, 0], [ 10, 2, 5],  [ 2, 5, 8],   [-2, 5, 8]],    color: WHT,  r: 0.40, op: 0.20 },
        // Optic radiation
        { pts: [[-12, 0, -14], [-6, -4, -8], [0, -8, -4], [4, -14, 4]],   color: EMR,  r: 0.38, op: 0.38 },
        { pts: [[ 12, 0, -14], [ 6, -4, -8], [0, -8, -4], [-4, -14, 4]],  color: EMR,  r: 0.38, op: 0.38 },
        // Arcuate fasciculus — arching language tract
        { pts: [[-20, 10, 5], [-14, 18, 0], [-8, 14, -8], [-6, 4, -12]],  color: CYN,  r: 0.45, op: 0.32 },
        { pts: [[ 20, 10, 5], [ 14, 18, 0], [ 8, 14, -8], [ 6, 4, -12]],  color: PUR,  r: 0.45, op: 0.32 },
        // Fornix arching over corpus callosum
        { pts: [[-8, 8, -2], [-4, 12, 0], [0, 14, 2], [4, 12, 0], [8, 8, -2]], color: VIO, r: 0.50, op: 0.28 },
    ];

    tractDefs.forEach(td => {
        const cv  = new THREE.CatmullRomCurve3(td.pts.map(p => new THREE.Vector3(...p)));
        root.add(new THREE.Mesh(
            new THREE.TubeGeometry(cv, 24, td.r, 7, false),
            new THREE.MeshBasicMaterial({ color: td.color, transparent: true, opacity: td.op })
        ));
    });

    /* ═══════════════════════════════════════════════════════
       SYNAPTIC FIRE PARTICLES
       300 bright dots randomly distributed across cortex surface,
       they flicker on/off simulating neural firing
    ═══════════════════════════════════════════════════════ */
    const SYN_N  = 340;
    const synPos = new Float32Array(SYN_N * 3);
    const synPhi = new Float32Array(SYN_N);
    const synTht = new Float32Array(SYN_N);
    const synSide= new Uint8Array(SYN_N);  // 0=left, 1=right
    const synLife= new Float32Array(SYN_N);

    for (let i = 0; i < SYN_N; i++) {
        synSide[i] = i % 2;
        synPhi[i]  = 0.2 + Math.random() * (Math.PI - 0.4);
        synTht[i]  = synSide[i] === 0
            ? 0.05 + Math.random() * (Math.PI - 0.1)
            : Math.PI + 0.05 + Math.random() * (Math.PI - 0.1);
        synLife[i] = Math.random();
        const a = synSide[i] === 0 ? LA : RA;
        const b = synSide[i] === 0 ? LB : RB;
        const c = synSide[i] === 0 ? LC : RC;
        const ox = synSide[i] === 0 ? LOFF : ROFF;
        const p = ellipsoidPt(a, b, c, synPhi[i], synTht[i]);
        synPos[i*3] = p.x + ox; synPos[i*3+1] = p.y; synPos[i*3+2] = p.z;
    }

    const synGeo = new THREE.BufferGeometry();
    synGeo.setAttribute('position', new THREE.BufferAttribute(synPos, 3));
    const synMat = new THREE.PointsMaterial({
        color: WHT, size: 1.8, transparent: true, opacity: 0.80,
        sizeAttenuation: true, depthWrite: false,
    });
    root.add(new THREE.Points(synGeo, synMat));

    /* ═══════════════════════════════════════════════════════
       TRAVELLING SIGNAL DOTS along tracts
    ═══════════════════════════════════════════════════════ */
    const SIG_N   = 120;
    const sigPos  = new Float32Array(SIG_N * 3);
    const sigT    = new Float32Array(SIG_N);
    const sigCurves = tractDefs.map(td =>
        new THREE.CatmullRomCurve3(td.pts.map(p => new THREE.Vector3(...p)))
    );
    for (let i = 0; i < SIG_N; i++) {
        sigT[i] = Math.random();
        const p = sigCurves[i % sigCurves.length].getPoint(sigT[i]);
        sigPos[i*3] = p.x; sigPos[i*3+1] = p.y; sigPos[i*3+2] = p.z;
    }
    const sigGeo = new THREE.BufferGeometry();
    sigGeo.setAttribute('position', new THREE.BufferAttribute(sigPos, 3));
    const sigMat = new THREE.PointsMaterial({
        color: CYN, size: 2.4, transparent: true, opacity: 0.90,
        sizeAttenuation: true, depthWrite: false,
    });
    root.add(new THREE.Points(sigGeo, sigMat));

    /* ═══════════════════════════════════════════════════════
       MRI SCAN RING — horizontal plane sweeping up/down
    ═══════════════════════════════════════════════════════ */
    const scanRing = new THREE.Mesh(
        new THREE.TorusGeometry(30, 0.18, 8, 100),
        new THREE.MeshBasicMaterial({ color: CYN, transparent: true, opacity: 0.35 })
    );
    scanRing.rotation.x = Math.PI / 2;
    root.add(scanRing);

    // Second outer ring — static orbit
    root.add(new THREE.Mesh(
        new THREE.TorusGeometry(38, 0.12, 6, 120),
        new THREE.MeshBasicMaterial({ color: PUR, transparent: true, opacity: 0.14 })
    ));

    /* ═══════════════════════════════════════════════════════
       DEEP STAR FIELD
    ═══════════════════════════════════════════════════════ */
    const STAR_N = 700;
    const starArr = new Float32Array(STAR_N * 3);
    for (let i = 0; i < STAR_N; i++) {
        starArr[i*3]   = (Math.random() - 0.5) * 700;
        starArr[i*3+1] = (Math.random() - 0.5) * 500;
        starArr[i*3+2] = -150 - Math.random() * 200;
    }
    const starGeo = new THREE.BufferGeometry();
    starGeo.setAttribute('position', new THREE.BufferAttribute(starArr, 3));
    scene.add(new THREE.Points(starGeo,
        new THREE.PointsMaterial({ color: TCYN, size: 0.55, transparent: true, opacity: 0.15, sizeAttenuation: true, depthWrite: false })
    ));

    /* ═══════════════════════════════════════════════════════
       ANIMATION LOOP
    ═══════════════════════════════════════════════════════ */
    let clock = 0;

    function animate() {
        requestAnimationFrame(animate);
        clock += 0.007;

        // Mouse lerp
        mouse.x += (target.x - mouse.x) * 0.030;
        mouse.y += (target.y - mouse.y) * 0.030;

        // Camera parallax
        camera.position.x += (mouse.x * 12 - camera.position.x) * 0.018;
        camera.position.y += (12 - mouse.y * 8 - camera.position.y) * 0.018;
        camera.lookAt(0, 4, 0);

        // Slow brain rotation + mouse tilt
        root.rotation.y += 0.0014;
        root.rotation.x  = mouse.y * 0.10;
        root.rotation.z  = mouse.x * 0.05;

        // MRI scan ring sweeps vertically
        scanRing.position.y      = Math.sin(clock * 0.45) * 22;
        scanRing.material.opacity = 0.18 + 0.22 * Math.abs(Math.sin(clock * 0.45));

        // Neural pulse — 60 BPM rhythm drives synapse firing
        const beat    = Math.sin(clock * 1.28);
        const fire    = Math.max(0, beat);

        // Synaptic particles flicker
        if (Math.floor(clock * 60) % 3 === 0) {
            for (let i = 0; i < SYN_N; i++) {
                synLife[i] += 0.06 + Math.random() * 0.04;
                if (synLife[i] > 1) synLife[i] = 0;
            }
        }
        synMat.opacity = 0.50 + fire * 0.42;
        synMat.size    = 1.4 + fire * 0.7;

        // Signal dots travel along tracts
        const SP = sigGeo.attributes.position;
        const spd = 0.004 + fire * 0.003;
        for (let i = 0; i < SIG_N; i++) {
            sigT[i] += spd * (0.5 + (i % 5) * 0.18);
            if (sigT[i] > 1) sigT[i] -= 1;
            const p = sigCurves[i % sigCurves.length].getPoint(Math.min(sigT[i], 0.999));
            SP.array[i*3] = p.x; SP.array[i*3+1] = p.y; SP.array[i*3+2] = p.z;
        }
        SP.needsUpdate = true;
        sigMat.opacity = 0.60 + fire * 0.35;
        sigMat.size    = 2.0 + fire * 1.0;

        // Cerebellum gentle sway
        cbGroup.rotation.z = Math.sin(clock * 0.38) * 0.03;

        renderer.render(scene, camera);
    }

    animate();

    window.addEventListener('resize', () => {
        camera.aspect = window.innerWidth / window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
    });

    window.addEventListener('scroll', () => {
        const t = window.scrollY / Math.max(1, document.body.scrollHeight - window.innerHeight);
        root.position.y = -t * 16;
    });

})();

}
